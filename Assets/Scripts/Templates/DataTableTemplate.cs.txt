//   2. 用户在 Unity 菜单 Tools → DataTable → 生成全部配置表代码

//
//   3. Generator 输出：
//      - Assets/Scripts/DataTable/<Name>.cs            （含 <Name>Row + <Name> : IDataTable）
//      - Assets/Scripts/DataTable/DataTableRegistry.cs（自动更新注册）
//
// 支持的字段类型：int / float / double / string / bool / T[] / Dictionary<K, V>（K 必须基础类型）

using System.Collections.Generic;
using Newtonsoft.Json;

/// <summary>
/// <auto-generated>
/// 由 DataTableGenerator 生成，请勿手动修改
/// </auto-generated>
/// </summary>
public sealed class FooConfigRow
{
    /// <summary>行主键，对应 JSON fields[0]，强制为 int</summary>
    public int Id { get; set; }

    /// <summary>对应 JSON 中的 Name 字段</summary>
    public string Name { get; set; }

    /// <summary>对应 JSON 中的某个数值字段</summary>
    public float Value { get; set; }
}

public sealed class FooConfig : IDataTable
{
    readonly Dictionary<int, FooConfigRow> _rows = new();

    public void Load(string json)
    {
        var file = JsonConvert.DeserializeObject<DataTableFile<FooConfigRow>>(json);
        _rows.Clear();
        foreach (var row in file.rows)
        {
            _rows[row.Id] = row;
        }
    }

    public FooConfigRow GetById(int id)
    {
        if (_rows.TryGetValue(id, out var row))
            return row;

        FrameworkLogger.Error("FooConfig", $"Action=GetById Id={id} Result=NotFound");
        return null;
    }

    public bool TryGetById(int id, out FooConfigRow row) => _rows.TryGetValue(id, out row);

    public IReadOnlyDictionary<int, FooConfigRow> All => _rows;
}
