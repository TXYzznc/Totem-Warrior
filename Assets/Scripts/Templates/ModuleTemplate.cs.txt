// 2. 重命名为 <YourModule>.cs（去掉 .txt 后缀）

// 4. 根据需要填充初始化逻辑与事件处理器
// ================================================================

using System;
using System.Threading;
using Cysharp.Threading.Tasks;

public class YourModule : IGameModule
{
    // ========== 模块元数据 ==========

    /// <summary>
    /// 模块功能类别：
    /// 0 = 启动基础层（ConfigModule / ResourceModule）
    /// 1 = 运行服务层（AudioModule / NetworkModule）
    /// 2 = 应用协调层（UIModule / InputModule / CameraModule）
    /// 3 = 项目扩展层（CombatModule / QuestModule / PlayerModule）—— 默认
    /// 4 = 后台辅助层（AnalyticsModule / CrashReportModule）
    /// </summary>
    public int ModuleCategory => 3;

    /// <summary>
    /// 硬依赖。必须放具体的 Module 类型，不支持接口。
    /// ModuleRunner 保证这些模块初始化完成后才开始本模块的 InitializeAsync。
    /// 没有依赖时返回 Type.EmptyTypes。
    /// </summary>
    public Type[] Dependencies => new[] { typeof(SomeDependencyModule) };

    // ========== 外部模块引用缓存 ==========

    readonly ModuleRunner _runner;
    SomeDependencyModule _dependency;

    public YourModule(ModuleRunner runner)
    {
        _runner = runner ?? throw new ArgumentNullException(nameof(runner));
    }

    // ========== 初始化 ==========

    /// <summary>
    /// 模块初始化。Dependencies 保证已完成初始化。
    /// ⚠️ InitAsync 期间不发布事件；时序通信用 Dependencies 表达。
    /// </summary>
    public async UniTask InitializeAsync(CancellationToken ct = default)
    {
        // 1. 缓存依赖模块引用（必须是 Dependencies 中列出的模块）
        _dependency = _runner.GetModule<SomeDependencyModule>();

        // 2. 加载配置、初始化内部状态
        // await LoadConfigAsync(ct);

        await UniTask.CompletedTask;
        FrameworkLogger.Info("YourModule", "Action=Initialized");
    }

    // ========== 关闭 ==========

    /// <summary>
    /// 模块关闭。按初始化完成序的逆序调用。
    /// ShutdownAsync 期间可以发布事件（注销在 ShutdownAsync 之后）。
    /// </summary>
    public async UniTask ShutdownAsync(CancellationToken ct = default)
    {
        // 清理订阅、保存数据、释放资源
        // await SaveDataAsync(ct);

        await UniTask.CompletedTask;
        FrameworkLogger.Info("YourModule", "Action=Shutdown");
    }

    // ========== 事件处理器 ==========

    /// <summary>异步事件处理。InitAsync 完成后自动注册，ShutdownAsync 完成后自动注销。</summary>
    [EventHandler]
    async UniTask OnSomethingHappened(SomethingHappenedEvent evt)
    {
        // 异步处理事件
        await UniTask.CompletedTask;
    }

    /// <summary>同步事件处理。</summary>
    [EventHandler]
    void OnAnotherThing(AnotherThingEvent evt)
    {
        // 立即更新状态
    }
}

// ========== 占位类型（实际项目应在对应位置定义）==========

public class SomethingHappenedEvent { }
public class AnotherThingEvent { }

public class SomeDependencyModule : IGameModule
{
    public int ModuleCategory => 0;
    public Type[] Dependencies => Type.EmptyTypes;
    public UniTask InitializeAsync(CancellationToken ct = default) => UniTask.CompletedTask;
    public UniTask ShutdownAsync(CancellationToken ct = default) => UniTask.CompletedTask;
}
